Table ZRB_GRP as ZRB {
  CODIGO varchar [pk, increment] // auto-increment

}

// Creating tables
Table ZRC_ITEMGRP as ZRC {
  COD_ZRB varchar [pk, increment] // auto-increment

}

  CODIGO varchar [pk]

 }
